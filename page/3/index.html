算法基础 1. 数组的二分查找

Semones published on 2022-03-11 included in 算法基础

每次与中间元素比较，根据比较结果决定继续搜索左半部分还是右半部分。

1. 基本要求

  1. 返回数组中的指定的值的下标。
  2. 限定无重复元素的升序数组。

2. 代码实现

  1. 迭代
public class BinarySearchIteration {
	// 迭代
	public static int getIndex(int[] nums, int target) {
		// 检测目标是否在数组范围内
		if (target < nums[0] || target > nums[nums.length - 1]) {
            return -1;
        }

        // 在闭区间定义
		int left = 0;
		int right = nums.length - 1;

		// 闭区间允许 left == right,所以用 <=
		while (left <= right) {
			// 计算中点，如果是 (left + right) / 2，两数的和可能超过整型上限
			int mid = left + (right - left) / 2;

			if (target == nums[mid]) {
				return mid;
			} else if (target < nums[mid]) {
				right = mid - 1;
			} else {
				left = mid + 1;
			}
		}

		return -1;
	}
}
  1. 递归
public class BinarySearchRecursive {
	// 递归
	public static int getIndex(int[] nums, int target, int left, int right) {
		// 递归的出口
		if (left > right) {
			return -1;
		}
		// 检测目标是否在数组范围内
		if (target < nums[0] || target > nums[nums.length - 1]) {
			return -1;
		}

		int mid = left + (right - left) / 2;

		if (nums[mid] == target) {
            return mid;
          // 与迭代同理，落在左边就移动右指针，落在右边就移动左指针
        } else if (target < nums[mid]) {
            return getIndex(nums, target, left, mid - 1);
        } else {
            return getIndex(nums, target, mid + 1, right);
        }
	}
}
