算法基础 6. 设计链表

Semones published on 2022-06-30 included in 算法基础

利用方法扩展链表功能。

1. 基本要求

  1. 实现单双链表的功能：get(index) addAtHead(val) addAtTail(val) addAtIndex(index,val) deleteAtIndex(index)。
  2. 使用虚拟头节点，index的值与size同步。

2. 代码实现

  1. 单链表
public class SinglyLinkedList {
	// 单链表
	class ListNode {
		int val;
		ListNode next;
		ListNode(int val) {
			this.val = val;
		}
	}

	private int size;
	private ListNode head;

	// 初始化链表
	public SinglyLinkedList() {
		this.size = 0;
		this.head = new ListNode(0);
	}

	public int get(int index) {
		if (index < 1 || index > size) {
			return -1;
		}

		ListNode cur = head;
		// i 与真实节点的 index 同步
		for (int i = 1; i <= index; i++) {
			cur = cur.next;
		}

		return cur.val;
	}

	public void addAtHead(int val) {
		// 调用此方法时才会动态创建 newNode
		ListNode newNode = new ListNode(val);
		newNode.next = head.next;
		head.next = newNode;
		
		size++;
	}

	public void addAtTail(int val) {
		ListNode newNode = new ListNode(val);
		
		ListNode cur = head;
		while (cur.next != null) { // 此时 cur.next == null，也就是 cur 指向尾结点
			cur = cur.next;
		}
		cur.next = newNode;
		
		size++;
	}

	// 在 index 之前添加结点，添加后新结点的序号就是 index
	public void addAtIndex(int index, int val) {
		if (index < 1 || index > size + 1) { // 添加到末尾时，需要一个比 size 大 1 的虚拟 index
			return;
		}

		// 寻找要插入结点的前驱
		ListNode pre = head;
		for (int i = 1; i <= index - 1; i++) {
			pre = pre.next;
		}

		ListNode newNode = new ListNode(val);
		newNode.next = pre.next;
		pre.next = newNode;
		
		size++;
	}

	public void deleteAtIndex(int index) {
		if (index < 1 || index > size) {
			return;
		}

		// 寻找要插入结点的前驱
		ListNode pre = head;
		for (int i = 1; i <= index - 1; i++) {
			pre = pre.next;
		}
		pre.next = pre.next.next;
		
		size--;
	}

	public void printList() {
		ListNode cur = this.head.next;
		while (cur != null) {
			System.out.print(cur.val + " ");
			cur = cur.next;
		}

		System.out.println();
	}
}
  1. 双链表
public class DoublyLinkedList {
	// 双链表
	class ListNode {
		int val;
		ListNode prev, next;
		ListNode(int val) {
			this.val = val;
		}
	}

	private int size;
	// 虚拟头尾结点
	private ListNode head, tail;

	public DoublyLinkedList() {
		this.size = 0;
		this.head = new ListNode(0);
		this.tail = new ListNode(0);
		this.head.next = tail;
		this.tail.prev = head;
	}

	public int get(int index) {
		if (index < 1 || index > size) {
			return -1;
		}

		ListNode cur = head;
		// 判断从哪一边遍历时间短
		if (index <= size / 2) {
			for (int i = 1; i <= index; i++) {
				cur = cur.next;
			}
		} else {
			cur = tail;
			for (int i = size; i >= index; i--) {
				cur = cur.prev;
			}
		}

		return cur.val;
	}

	public void addAtHead(int val) {
		addAtIndex(1, val);
	}

	public void addAtTail(int val) {
		addAtIndex(size + 1, val);
	}

	public void addAtIndex(int index, int val) {
		if (index < 1 || index > size + 1) { // 添加到末尾时，需要一个比 size 大 1 的虚拟 index
			return;
		}

		// 寻找要插入结点的前驱
		ListNode pre = head;
		for (int i = 1; i <= index - 1; i++) {
			pre = pre.next;
		}

		ListNode newNode = new ListNode(val);
		newNode.prev = pre;
		newNode.next = pre.next; // 先把新结点与前驱结点和原结点相连
		pre.next.prev = newNode;
		pre.next = newNode; // 断开前驱结点与原结点的双向连接，分别与新结点相连

		size++;
	}

	public void deleteAtIndex(int index) {
		if (index < 1 || index > size) {
			return;
		}

		// 寻找要删除结点的前驱
		ListNode pre = head;
		for (int i = 1; i <= index - 1; i++) {
			pre = pre.next;
		}
		// 依赖于 pre 的定位，所以要由远及近修改
		pre.next.next.prev = pre; // 修改后继结点的 prev
		pre.next = pre.next.next; // 修改前驱结点的 next

		size--;
	}

	public void printList() {
		ListNode cur = this.head.next;
		while (cur != tail) {
			System.out.print(cur.val + " ");
			cur = cur.next;
		}

		System.out.println();
	}
}

算法基础 5. 前缀和

Semones published on 2022-05-28 included in 算法基础

前缀和 (Prefix Sum) 数组的第 i 个元素表示原数组从第 0 个元素到第 i 个元素的累加和。

1. 基本要求

  1. 计算数组在指定区间的和。
  2. 先输入数组长度，再依次输入数组元素，接着输入要求的区间。

2. 代码实现

/*
  1. 如果暴力求解，查询 m 次，每次查询范围都是 0 到 n - 1，时间复杂度为 O(n * m)，会非常大。
  2. 前缀和可以把时间复杂度降为 O(1)。
  3. Java 控制台中，空格或回车分隔数据。一次性输入大量数据后，按照代码顺序依次读取。
*/

import java.util.Scanner;

public class IntervalSum{
	// 前缀和
	public static void main(String[] args) {
		Scanner scanner = new Scanner(System.in);

		int n = scanner.nextInt();
		int[] nums = new int[n];
		// prefix 指开头到某个位置的连续部分。
		int prefixSum = 0; // 前缀和指数组中前 i 个元素的总和
		int[] prefixSumArr = new int[n]; // 前缀和的数组

		// 构建原数组和前缀和数组
		for (int i = 0; i < n; i++) {
			nums[i] = scanner.nextInt();
			prefixSum += nums[i];
			prefixSumArr[i] = prefixSum;
		}

		// hasNextInt() 检查下一个输入是否为整数，但不会实际读取
		// 实现未知数量输入处理
		while (scanner.hasNextInt()) {
			int a = scanner.nextInt();
            int b = scanner.nextInt();
            int sum;

            if (a == 0) { // 等价于前缀和
                sum = prefixSumArr[b];
            } else {
            	// 用前缀和求区间和
                sum = prefixSumArr[b] - prefixSumArr[a - 1];
            }
            
            System.out.println(sum);
		}

		// 关闭 Scanner 对象释放资源。
		scanner.close();
	}
}

算法基础 4. 生成螺旋矩阵

Semones published on 2022-05-14 included in 算法基础

利用指针和偏移量达到螺旋效果。

1. 基本要求

  1. 按顺时针顺序生成 1 到 n^2 螺旋排列的正方形矩阵。
  2. 输出二维数组：
for (int[] row : array) {
	System.out.println(Arrays.toString(row));
}

2. 代码实现

public class SpiralMatrix{

	public int[][] getMatrix(int n) {
		int[][] matrix = new int[n][n];
		int startR = 0, startC = 0; // 行与列的序号构成每圈的起始点
		int offset = 1; // 左闭右开偏移量
		int count = 1; // 待填写的数字
		int loop = 1; // 当前圈数
		int i, j; // i 指向每一行，纵向移动；j 指向每一列，横向移动

		// 构建行时，要用指向列的 j 扫描；构建列时，要用指向行的 i 扫描
		// i 与 j 分别停止等待对方走完列或行，接力转圈。
		while (loop <= n / 2) { // 起始点从左上角沿对角线向中心靠拢，到达中心点前停止
			for (j = startC; j < n - offset; j++) {
                matrix[startR][j] = count++;
            }
            for (i = startR; i < n - offset; i++) {
            	matrix[i][j] = count++;
            }
            for (; j > startC; j--) {
            	matrix[i][j] = count++;
            }
            for (; i > startR; i--) {
            	matrix[i][startC] = count++;
            }

            // 起始点自增以缩圈
            startR++;
            startC++;
            offset++;
            loop++;
		}

		// 跳出 wihle 循环时，startR 与 startC 一定比 n / 2 大 1
		// 如果矩阵边长时奇数，会留下一个中心点
		if (n % 2 == 1) {
			matrix[startR][startC] = count;
		}

		return matrix;
	}
}

算法基础 3. 滑动窗口寻找子数组

Semones published on 2022-04-06 included in 算法基础

滑动窗口就是不断调节子序列的起始位置和终止位置，从而得到想要的结果。

1. 基本要求

  1. 寻找给定数组中长度最短的子数组，其元素和大于等于给定的数，返回长度。

2. 代码实现

public class MinSubArrayLen {
	// 滑动窗口
	public int getLength(int[] nums, int s) {
		int left = 0;
		int sum = 0;
		int length = Integer.MAX_VALUE;

		// 每轮 for 循环都用 sum 对比给定值判断是否进入 while 循环刷新长度
		// 跳出 while 循环继续 for 循环时，sum 一定小于给定值，此时右指针自增继续
		for (int right = 0; right < nums.length; right++) {
			sum += nums[right];

			// 当 sum 超过给定值时，可能远大于此值，所以要尝试减去左边的值再次比较
			while (sum >= s) {
				// 先记录长度，再尝试缩小子数组
				length = Math.min(length, right - left + 1);
				sum -= nums[left++];
			}
		}

		// 三元运算符，如果结果没改变，说明不存在子数组的和大于给定值
		return length == Integer.MAX_VALUE ? 0 : length;
	}
}

算法基础 2. 双指针移除数组元素

Semones published on 2022-03-23 included in 算法基础

使用两个指针配合，在一个 for 循环下完成两个 for 循环的工作。

1. 基本要求

  1. 移除指定元素，不使用额外数组空间，可改变顺序。函数返回新的长度。
  2. 数组的元素在内存地址中是连续的，不能单独删除数组中的某个元素，只能覆盖。

2. 代码实现

  1. 快慢指针
public class RemoveElementsFastSlowPointer {
	// 快慢指针
	public int getLength(int[] nums, int val) {
		int slow = 0;

		// 快指针遍历数组
		for (int fast = 0; fast < nums.length; fast++) {
			// 快指针扫描到目标元素时不给慢指针赋值，慢指针停留等待快指针扫描非目标元素来完成覆盖
			if (nums[fast] != val) {
				nums[slow] = nums[fast];
				slow++;
			}
		}

		// fast 每扫描到一个非目标元素，slow 自增，恰好是新数组长度
		return slow;
	}
}
  1. 左右指针
public class RemoveElementsLeftRightPointer {
	// 左右指针
	public int getLength(int[] nums, int val) {
        // 两指针从两边向中间靠拢
		int left = 0;
        int right = nums.length - 1;

        // 左指针自增，扫描到目标值用右指针的值覆盖后再次比较，事实上每个值都靠 left 比较
        while (left <= right) { // left == right 时，比较最后一个元素
            if (nums[left] == val) {
                nums[left] = nums[right];
                right--;
            } else { // 确保左指针的值不等于目标值，即使右指针也恰好把目标值赋给了左指针，也不会自增
                left++;
            }
        }

        // 只有非目标元素才会使 left 自增
        return left;
	}
}
