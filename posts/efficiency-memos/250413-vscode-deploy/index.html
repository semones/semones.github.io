🛠️ 全局设置

1. 程序设置

// 程序设置
"update.enableWindowsBackgroundUpdates": false,	// 禁止后台下载更新
"update.mode": "none",							// 禁止任何更新行为
"extensions.autoCheckUpdates": false,           // 禁止插件检查更新
"extensions.autoUpdate": false,                 // 禁止插件自动更新
"extensions.ignoreRecommendations": true,		// 关闭插件安装建议
"files.autoGuessEncoding": true,				// 打开文件时猜测字符集编码
"explorer.compactFolders": false,				// 关闭文件夹紧凑显示模式
"chat.agentsControl.enabled": "hidden",			// 关闭 Copilot 代理控制功能

2. 工作区设置

// 工作区设置
"workbench.startupEditor": "none",				// 关闭启动页面
"workbench.editor.empty.hint": "hidden",		// 编辑器为空时，不提示打开文件
"workbench.list.smoothScrolling": true,			// 开启界面平滑滚动效果
"workbench.colorTheme": "Dracula Theme Soft",	// 用 Dracula Theme Official 插件设置主题外观
"workbench.iconTheme": "material-icon-theme",	// 用 Material Icon Theme 插件设置图标样式

3. 编辑器设置

// 编辑器设置
"editor.fontSize": 18,							// 字体大小
"editor.mouseWheelZoom": true,					// 允许使用 ctrl + 滚轮缩放字体大小
"editor.fontFamily": "Consolas, '微软雅黑'",	 // 设置字体
"editor.smoothScrolling": true,					// 开启编辑器区域平滑滚动效果
"editor.cursorSmoothCaretAnimation": "on",		// 开启光标平滑移动效果
"editor.cursorBlinking": "smooth",				// 开启光标淡入淡出效果
"editor.fontLigatures": false,					// 关闭字体连字效果

4. 终端设置

// 终端设置
"terminal.integrated.smoothScrolling": true,					// 开启终端平滑滚动效果
"terminal.integrated.defaultProfile.windows": "PowerShell",		// 设置终端默认启动 PowerShell
"terminal.integrated.profiles.windows": {						// 终端详细配置
    "PowerShell": {
        "path": "C:\\Windows\\System32\\WindowsPowerShell\\v1.0\\powershell.exe"
    },
    "Command Prompt": {
        "path": "C:\\Windows\\System32\\cmd.exe",
        "args": [],
        "icon": "terminal-cmd"
    },
    "Git Bash": {
        "source": "Git Bash",
        "icon": "terminal-git-bash"
    }
},

🔧 插件配置

1. Code Runner

// Code Runner
"code-runner.runInTerminal": true,					// 在 VS Code 集成的终端中运行
"code-runner.languageIdToFileExtensionMap": {		// 扩展名映射
    "bat": ".bat",
    "powershell": ".ps1",
    "typescript": ".ts"
},
"code-runner.executorMap": {						// 编程语言对应的执行命令
    "javascript": "node",
    "php": "C:\\php\\php.exe",
    "python": "python",
    "perl": "perl",
    "ruby": "C:\\Ruby23-x64\\bin\\ruby.exe",
    "go": "go run",
    "html": "\"C:\\Program Files (x86)\\Google\\Chrome\\Application\\chrome.exe\"",
    "java": "cd $dir && javac $fileName && java $fileNameWithoutExt",
    // 控制台代码页切换到 UTF-8 (65001)，避免中文乱码
    "c": "chcp 65001 && cd $dir && gcc $fileName -o $fileNameWithoutExt && $dir$fileNameWithoutExt"
},

2. Lingma - Alibaba Cloud AI Coding Assistant

// Lingma - Alibaba Cloud AI Coding Assistant
"Lingma.aI Chat.memories": false,							// 关闭通义灵码记忆功能
"Lingma.aI Chat.webToolsInAsk/AgentMode": "Auto-execute",	// 允许通义灵码自动调用工具

3. Python

// Python
"python.autoComplete.extraPaths": [],				// 为自动补全功能添加额外的模块搜索路径
"python.analysis.completeFunctionParens": true,		// 开启自动补全括号和占位符
"python.analysis.typeCheckingMode": "standard",		// 设置类型检查模式为标准
